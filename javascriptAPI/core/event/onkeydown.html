onkeydown

描述

键码：oEvent.keyCode<数字>

* 在js里面，键码不能组合来用的

oEvent.ctrlKey ctrl键

oEvent.shiftKey shift键

oEvent.altKey alt键

示例

描述

组合键使用方式。

js代码：
document.onkeydown=function(ev){
	var oEvent=ev || event;
	//ctrl  17
	//c   67
	//s   83
	if(oEvent.ctrlKey && oEvent.keyCode==67){
		alert('复制了');	
	}
	if(oEvent.ctrlKey && oEvent.shiftKey && oEvent.altKey && oEvent.keyCode==83){
		alert('保存为web格式');
	}
};
