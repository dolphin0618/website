this优先级：
function show()
{
	alert(this);
}

var arr=[1,2,3];
arr.show=show;

show();			//window
arr.show();		//arr

new show();	//空白对象
new arr.show();	//空白对象

document.onclick=show;		//document
document.onclick=arr.show;	//document


document.onclick=show;
document.onclick();				//document


document.onclick=arr.show;
document.onclick();				//document


document.onclick=show;
new document.onclick();			//空白对象


document.onclick=arr.show;
new document.onclick();			//空白对象


setTimeout(show, 10);			//window
setTimeout(arr.show, 10);		//window


document.onclick=show;
setTimeout(document.onclick, 10);	//window


document.onclick=arr.show;
setTimeout(document.onclick, 10);	//window


document.onclick=arr.show;
setTimeout(new document.onclick(), 10);	//空白对象
