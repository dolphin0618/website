文本选中与光标定位

描述

用js来选中文字和定位光标

示例

描述

设置输入框选中文字方式：
if(oT.setSelectionRange){
	oT.setSelectionRange(开始位置,结束位置); 兼容IE9+ chrome FF //selectionStart
	oT.focus();	
}else{ 
	var range=oT.createTextRange(); 
	range.collapse(false); //规定选文本方向 
	range.moveStart('character',开始位置); 
	range.moveEnd('character',结束位置); 
	range.select(); //选择 
	range.text.length;//选中文字的长度
	oT.focus(); 
}
//如果开始位置和位置相等，就等于定位光标
弹出选中的文字：
oP.onmouseup=function(){
	if(document.getSelection){
		var text=document.getSelection();
		alert(text);
	}else{
		var range=document.selection.createRange();
		alert(range.text);
	}
};
